lass Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n=nums.size();
        int s=0,e=n-1,m=s+((e-s)/2);
        
        while(s<=e){
            if(nums[m]==target){
                return m;
            }
            else if(nums[m]<target)
                s=m+1;
            else
                e=m-1;
            m=s+((e-s)/2);
        }
        return s;
        
    }
};
